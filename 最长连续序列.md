public class LongestSerial {
    int longestSerial(int[] nums) {
        Arrays.sort(nums);
        int temp = 1;
        int maxLength = 1;

        if (nums.length == 0) {
            return 0;
        }
        for (int i = 1; i < nums.length; i++) {
            // 重复的元素
            if (nums[i] == nums[i - 1]) {
                continue;
            }
            if (nums[i] - nums[i - 1] == 1) {
                temp = temp + 1;
                maxLength = Math.max(temp, maxLength);
            } else {
                temp = 1;
            }
        }
        return maxLength;
    }
}
